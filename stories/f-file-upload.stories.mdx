<Canvas>
	<Story
		name="Playground"
		argTypes={{
			value: { control: "object" },
			type: { control: "radio", options: ["single", "multiple"] },
			placeholder: { control: "text" },
			state: {
				control: "select",
				options: ["default", "success", "primary", "warning", "danger"]
			},
			["file-type"]: {
				control: "text"
			},
			size: {
				control: "radio",
				options: ["small", "medium"]
			},
			["max-size"]: {
				control: "text"
			},
			slot: { control: "text" },
			disabled: {
				control: "boolean"
			},
			loading: {
				control: "boolean"
			}
		}}
		args={{
			value: undefined,
			type: "single",
			placeholder: "Drag and Drop Files or Click here to upload",
			state: "default",
			size: "medium",
			["file-type"]: "all",
			["max-size"]: undefined,
			slot: `              <f-div slot="label" padding="none" gap="none">Label</f-div>
              <f-text slot="help" variant="para" size="small">This is a Subtext</f-text>`,
			disabled: false,
			loading: false
		}}
	>
		{args => {
			const [_, updateArgs] = useArgs();
			const handleInput = e => {
				if (e?.detail) {
					console.log(e.detail.value);
					if (args.type === "single") {
						updateArgs({ value: e.detail.value ? { name: e.detail.value.name } : e.detail.value });
					} else {
						updateArgs({ value: e.detail.value });
					}
				}
			};
			return html`
				<f-div width="100%" padding="large">
					<f-div width="80%">
						<f-file-upload
							.type=${args.type}
							.value=${args.value}
							placeholder=${args.placeholder}
							.state=${args.state}
							.size=${args.size}
							file-type=${args["file-type"]}
							max-size=${args["max-size"]}
							?disabled=${args.disabled}
							?loading=${args.loading}
							@input=${handleInput}
						>
							${unsafeHTML(args.slot)}
						</f-file-upload>
					</f-div></f-div
				>
			`;
		}}
	</Story>
</Canvas>
