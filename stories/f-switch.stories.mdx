# f-switch

<br />
<br />

## Overview

<f-divider></f-divider>

f-switch allows a user to turn an individual option on/off. The change in selection takes place immediately unlike checkboxes and radio buttons.

<br />

### [Playground](/story/components-f-switch--playground)

<Canvas>
  <Story
    name="Playground"
    argTypes={{
      value: { control: "radio", options: ["on", "off"] },
      state: {
        control: "select",
        options: ["default", "success", "primary", "warning", "danger"],
      },
      size: {
        control: "radio",
        options: ["small", "medium"],
      },
      slot: { control: "text" },
      disabled: {
        control: "boolean",
      },
    }}
    args={{
      value: "on",
      state: "default",
      size: "small",
      slot: ` <f-div padding="none" align="middle-center" slot="label">
              <f-text variant="para" size="small">Switch</f-text>
            </f-div>
            <f-div padding="none" align="middle-center" slot="help">
              <f-text variant="para" size="small">This is helper text</f-text>
            </f-div>`,
      disabled: false,
    }}
  >
    {(args) => {
      const [_, updateArgs] = useArgs();
      const handleInput = (e) => {
        updateArgs({ value: e.detail.value });
      };
      return html`
        <f-div padding="medium">
          <f-switch
            size=${args.size}
            value=${args.value}
            state=${args.state}
            ?disabled=${args.disabled}
            @input=${handleInput}
          >
            ${unsafeHTML(args.slot)}
          </f-switch>
        </f-div>
      `;
    }}
  </Story>
</Canvas>
