# f-text-area

<br />
<br />

## Overview

<f-divider></f-divider>

f-text-area allows the user to input a longer amount of text than f-input. It is commonly useful for descriptions and user feedback.

<br />

  <Story name="Flags">
    {(args) => {
      const [value, setValue] = useState("");
      const handleValue = (e) => {
        setValue(e.detail.value);
      };
      return html`
        <f-div width="100%" align="top-center" padding="large" gap="medium">
          ${[0, 1, 2, 3].map(
            (item) => html`<f-div>
          <f-text-area
            value=${value}
            placeholder="Write here"
            @input=${handleValue}
            size="medium"
            ?resizable=${item === 0 ? true : false}
            ?disabled=${item === 1 ? true : false}
            ?clear=${item === 2 ? true : false}
            ?read-only=${item === 3 ? true : false}
          >
            <f-div slot="label" padding="none" gap="none">${
              item === 0
                ? "Resizable"
                : item === 1
                ? "Disabled"
                : item === 2
                ? "Clear Icon on Type"
                : "Readonly"
            }</f-div>
            <f-div slot="description" padding="none" gap="none">This is a ${
              item === 0
                ? "Resizable"
                : item === 1
                ? "Disabled"
                : item === 2
                ? "Clear Icon on Type"
                : "Readonly"
            } Flag</f-div>
            <f-text slot="help" variant="para" size="small">This is a Subtext (Helper Text)</f-text>
          </f-text-area></f-div
